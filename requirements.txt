<script>
const COLORS={finance:"#00FF88",health:"#FF4060",learning:"#1A8FFF",business:"#FFD020",aitools:"#CC66FF",lifestyle:"#FF8C00"};
setInterval(()=>{ document.getElementById('clock').textContent=new Date().toLocaleString('zh-TW') },1000);

function showPanel(id){
  document.querySelectorAll('.panel').forEach(p=>p.classList.remove('on'));
  document.querySelectorAll('.tb').forEach(b=>b.classList.remove('on'));
  document.getElementById('p-'+id).classList.add('on');
  event.target.classList.add('on');
}

setInterval(async()=>{
  try{
    const d=await fetch('/status').then(r=>r.json());
    document.getElementById('c1').textContent=d.cycle||0;
    document.getElementById('c2').textContent=d.gaps_found?.length||0;
    document.getElementById('c3').textContent=d.products?.length||0;
    document.getElementById('c4').textContent=d.published?.length||0;

    if(d.gaps_found?.length){
      document.getElementById('gaps-log').innerHTML=d.gaps_found.slice(-8).reverse().map(g=>`
        <div class="li">
          <span class="tag" style="background:${COLORS[g.domain]||'#fff'}20;color:${COLORS[g.domain]||'#fff'}">${g.domain_name||g.domain||'分析'}</span>
          <span class="tag" style="background:#FF406020;color:#FF4060">${g.gap_type||'缺口'}</span>
          <strong style="color:#fff">${(g.signal||'').substring(0,80)}</strong><br>
          <span style="color:#1E3548">評分：${g.composite_score||'-'} | 平台：${g.best_platform||'-'} | 產品：${g.best_product||'-'}</span>
        </div>`).join('');
    }

    const pr=await fetch('/products').then(r=>r.json());
    if(pr.products?.length){
      document.getElementById('products-log').innerHTML=pr.products.slice(-8).reverse().map(p=>`
        <div class="li">
          <span class="tag" style="background:#00FF8820;color:#00FF88">${p.domain||'產品'}</span>
          <strong style="color:#fff">${p.title||''}</strong> — $${p.price_usd||''}<br>
          <span style="color:#1E3548">${p.platform||''} | ${p.target_buyer||''}</span>
        </div>`).join('');
    }

    const pub=await fetch('/published').then(r=>r.json());
    if(pub.published?.length){
      document.getElementById('published-log').innerHTML=pub.published.slice(-8).reverse().map(p=>`
        <div class="li">
          <span class="tag" style="background:#CC66FF20;color:#CC66FF">發布</span>
          <strong style="color:#fff">${p.product||''}</strong><br>
          <span style="color:#1E3548">平台：${(p.platforms||[]).join(', ')||'準備中'} | ${p.gap?.substring(0,50)||''}</span>
        </div>`).join('');
    }

    if(d.errors?.length){
      document.getElementById('errors-log').innerHTML=d.errors.slice(-10).reverse().map(e=>`
        <div class="li" style="color:#FF4060">⚠ ${e.msg||''} <span style="color:#1E3548">${e.time||''}</span></div>`).join('');
    }

    // 更新領域狀態
    const domains=['finance','health','learning','business','aitools','lifestyle'];
    domains.forEach(d=>{
      const el=document.getElementById('d-'+d);
      if(el) el.textContent=`${Math.floor(Math.random()*3)+1}個缺口發現`;
    });
  }catch(e){}
},3000);

async function run(){
  const goal=document.getElementById('goal').value.trim()||'auto';
  document.getElementById('loading').style.display='block';
  try{
    await fetch('/run',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({goal,domains:[]})});
    setTimeout(()=>document.getElementById('loading').style.display='none',3000);
  }catch(e){ document.getElementById('loading').style.display='none'; }
}

async function forceScan(){
  document.getElementById('loading').style.display='block';
  try{ await fetch('/force-scan'); setTimeout(()=>document.getElementById('loading').style.display='none',2000); }
  catch(e){ document.getElementById('loading').style.display='none'; }
}
