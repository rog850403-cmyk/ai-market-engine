def ai(engine,prompt,tokens=600):
    try:

                json={"model":"llama-3.3-70b-versatile","messages":[{"role":"user","content":prompt}],"max_tokens":tokens},timeout=25)

    except:

def think(prompt,lang="zh"):
    engines=["qwen","groq","gemini"] if lang=="zh" else ["groq","gemini","deepseek"]
    answers=[]
    for e in engines:
        r=ai(e,prompt)
        if r and len(r)>30:
            answers.append({"ai":e,"text":r})
        time.sleep(1)
    if not answers:return ""
    if len(answers)==1:return answers[0]["text"]
    merge=f"整合以下{len(answers)}個AI分析，輸出最準確答案，{'繁體中文' if lang=='zh' else 'English'}：\n"+"".join([f'[{a["ai"]}]:{a["text"][:300]}\n' for a in answers])
    m=ai("groq",merge,500)
    return m if m else answers[0]["text"]

def process(domain):
    name=domain["name"];lang=domain["lang"]
    print(f"  [掃描] {name}...")
    result={"domain_id":domain["id"],"domain_name":name,"status":"processing"}
    try:
        p=f"""分析「{name}」領域，種子：{'/'.join(domain['seeds'][:2])}
今天最熱的具體話題，輸出JSON（只輸出JSON，用{'繁體中文' if lang=='zh' else 'English'}）：
{{"topic":"具體話題一句話","pain_point":"核心痛點","emotion":"情緒","virality_score":0到100,"monetize_score":0到100,"composite_score":平均分}}"""
        raw=think(p,lang)
        try:
            clean=raw.strip()
            s=clean.find("{");e=clean.rfind("}")+1
            t=json.loads(clean[s:e])
            t["domain"]=name;t["lang"]=lang
            result["topic"]=t
            score=t.get("composite_score",50)
            print(f"  ✓ {name}: {t.get('topic','')[:30]} ({score}分)")
        except:
            result["topic"]={"topic":raw[:60],"composite_score":50,"domain":name,"lang":lang}
            score=50

        if score>=65:
            cp=f"""你是暗面筆記。風格：看穿別人沒說的那一面。
話題：{result['topic'].get('topic','')}
輸出JSON：{{"threads":"貼文200字繁中","telegram":"精華80字","hook":"開場15字內"}}"""
            raw2=think(cp,lang)
            try:
                clean2=raw2.strip();s=clean2.find("{");e=clean2.rfind("}")+1
                result["content"]=json.loads(clean2[s:e])
                result["action"]="publish"
            except:
                result["content"]={"threads":raw2[:200],"telegram":raw2[:80]}
                result["action"]="publish"
        else:
            result["action"]="skip"
            print(f"  ⚪ {name}: 分數不足({score})跳過")
        result["status"]="done"
    except Exception as ex:
        result["status"]="error";result["error"]=str(ex)
    return result

def run():
    start=datetime.now()
    print(f"\n{'='*50}")
    print(f"⬡ 全自主系統 {start.strftime('%Y/%m/%d %H:%M')}")
    print(f"  {len(DOMAINS)}個領域並行掃描")
    print(f"{'='*50}\n")
    results=[]
    with ThreadPoolExecutor(max_workers=3) as ex:
        futures={ex.submit(process,d):d for d in DOMAINS}
        for f in as_completed(futures):
            r=f.result()
            with LOCK:results.append(r)
    results.sort(key=lambda x:x.get("topic",{}).get("composite_score",0),reverse=True)
    print(f"\n{'─'*50}")
    print(f"★ 今日掃描結果")
    for r in results:
        t=r.get("topic",{})
        icon={"publish":"🟢","skip":"⚪","error":"🔴"}.get(r.get("action",""),"⚪")

    ts=start.strftime("%m%d_%H%M")
